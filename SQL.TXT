CREATE TABLE customer_message(timeadded TIMESTAMP DEFAULT CURRENT_TIMESTAMP, id INTEGER PRIMARY KEY AUTO_INCREMENT NOT NULL, name VARCHAR(255), email VARCHAR(255), phone VARCHAR(255), message VARCHAR(5000), comment VARCHAR(1000));

CREATE TABLE demo_request(timeadded TIMESTAMP DEFAULT CURRENT_TIMESTAMP, id INTEGER PRIMARY KEY AUTO_INCREMENT NOT NULL, name VARCHAR(255), email VARCHAR(255), phone VARCHAR(255), business_name VARCHAR(512), business_brief VARCHAR(2000));

CREATE TABLE career_request(timeadded TIMESTAMP DEFAULT CURRENT_TIMESTAMP, id INTEGER PRIMARY KEY AUTO_INCREMENT NOT NULL, name VARCHAR(255), email VARCHAR(255), phone VARCHAR(255), company VARCHAR(512), description VARCHAR(2000));

CREATE TABLE training_request(timeadded TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL, id INTEGER PRIMARY KEY AUTO_INCREMENT NOT NULL, name VARCHAR(255), email VARCHAR(255), phone VARCHAR(255), company VARCHAR(512), booking_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL, message VARCHAR(2000));
